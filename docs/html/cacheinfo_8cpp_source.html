45 // Helper function to count set bits in the processor mask.
46 DWORD CountSetBits(ULONG_PTR bitMask)
47 {
48  DWORD LSHIFT = sizeof(ULONG_PTR)*8 - 1;
49  DWORD bitSetCount = 0;
50  ULONG_PTR bitTest = (ULONG_PTR)1 << LSHIFT;
51  DWORD i;
52 
53  for (i = 0; i <= LSHIFT; ++i)
54  {
55  bitSetCount += ((bitMask & bitTest)?1:0);
56  bitTest/=2;
57  }
58 
59  return bitSetCount;
60 }

106  size_t line_size = 0;
107  DWORD buffer_size = 0;

109  SYSTEM_LOGICAL_PROCESSOR_INFORMATION * buffer = 0;
110 
111  GetLogicalProcessorInformation(0, &buffer_size);
112  buffer = (SYSTEM_LOGICAL_PROCESSOR_INFORMATION *)malloc(buffer_size);
113  GetLogicalProcessorInformation(&buffer[0], &buffer_size);
114 
115  for (i = 0; i != buffer_size / sizeof(SYSTEM_LOGICAL_PROCESSOR_INFORMATION); ++i)
116  {

121  }
